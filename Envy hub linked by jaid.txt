local function updateToggleVisual(indicator, state)
    if not indicator or not indicator.Parent then return end
    if state then
        indicator.BackgroundColor3          = Color3.fromRGB(0, 0, 0)
        indicator.Parent.BackgroundColor3   = Color3.fromRGB(255, 255, 255)
    else
        indicator.BackgroundColor3          = Color3.fromRGB(45, 45, 45)
        indicator.Parent.BackgroundColor3   = Color3.fromRGB(16, 16, 16)
    end
end

local function setToggle(name, state)
    toggles[name] = state
    local ind = toggleIndicators[name]
    if ind then updateToggleVisual(ind, state) end
end

local function readSpeed()
    local ns = tonumber(tbNormalSpeed  and tbNormalSpeed.Text)  or speedValues.NormalSpeed
    local cs = tonumber(tbCarrySpeed   and tbCarrySpeed.Text)   or speedValues.CarrySpeed
    local ls = tonumber(tbLaggerSpeed  and tbLaggerSpeed.Text)  or speedValues.LaggerSpeed
    local lcs= tonumber(tbLaggerCarrySpeed and tbLaggerCarrySpeed.Text) or speedValues.LaggerCarrySpeed
    return ns, cs, ls, lcs
end

local function currentSpeed()
    local ns, cs, ls, lcs = readSpeed()
    local m = modeIndex
    if     m == 4 then return lcs
    elseif m == 3 then return ls
    elseif m == 2 then return cs
    else               return ns end
end

local function applyDarkMode(enabled)
    if enabled then
        _defBright  = _defBright  or Lighting.Brightness
        _defClock   = _defClock   or Lighting.ClockTime
        _defAmbient = _defAmbient or Lighting.OutdoorAmbient
        Lighting.Brightness = 0; Lighting.ClockTime = 0
        Lighting.OutdoorAmbient = Color3.fromRGB(0,0,0)
        Lighting.GlobalShadows  = false
    else
        if _defBright  then Lighting.Brightness    = _defBright  end
        if _defClock   then Lighting.ClockTime      = _defClock   end
        if _defAmbient then Lighting.OutdoorAmbient = _defAmbient end
        Lighting.GlobalShadows = true
    end
end

-- ── Helpers ──────────────────────────────────────────────────
local function findBat()
    local c = Char(); local bp = LOCAL_PLAYER:FindFirstChild("Backpack")
    local function chk(t) local n = t.Name:lower()
        return n:find("bat") or n:find("slap") or n:find("club") or n:find("sword") end
    if c  then for _, t in ipairs(c:GetChildren())  do if t:IsA("Tool") and chk(t) then return t end end end
    if bp then for _, t in ipairs(bp:GetChildren()) do if t:IsA("Tool") and chk(t) then return t end end end
end

local function equip(tool)
    local h = Hum(); if h and tool then pcall(function() h:EquipTool(tool) end) end
end

local function swingBat()
    pcall(function()
        local c = Char(); if not c then return end
        local tool = c:FindFirstChildOfClass("Tool") or findBat()
        if not tool then return end
        if tool.Parent ~= c then equip(tool); task.wait(0.05) end
        pcall(function() tool:Activate() end)
        local gc = getconn()
        if gc then
            local ok, conns = pcall(gc, tool.Activated)
            if ok and conns then for _, cn in ipairs(conns) do if cn.Function then task.spawn(cn.Function) end end end
        end
    end)
end

local function instaReset()
    task.spawn(function()
        local ro = Root(); if ro then ro.CFrame = CFrame.new(0, 15000, 0) end
    end)
end

-- ── Speed loop (Heartbeat) ───────────────────────────────────
RunService.Heartbeat:Connect(function()
    if _jHolding and toggles.InfiniteJump and jumpMethod == "Hold" then
        local ro = Root()
        if ro then
            local v = ro.AssemblyLinearVelocity
            if v.Y < 55 then
                ro.AssemblyLinearVelocity = Vector3.new(v.X, 55, v.Z)
                pcall(function() ro.Velocity = Vector3.new(v.X, 55, v.Z) end)
            end
        end
    end

    local h  = Hum(); local ro = Root()
    if not h or not ro then return end
    if toggles.BatAimbot then return end
    if h.MoveDirection.Magnitude < 0.01 then return end

    -- Auto Switch Speed: switch to carry speed when near steal prompt
    if toggles.AutoSwitchSpeed then
        local plots = WS:FindFirstChild("Plots")
        local nearPrompt = false
        if plots then
            for _, plot in ipairs(plots:GetChildren()) do
                local pods = plot:FindFirstChild("AnimalPodiums")
                if pods then
                    for _, pod in ipairs(pods:GetChildren()) do
                        local base = pod:FindFirstChild("Base")
                        local sp   = base and base:FindFirstChild("Spawn")
                        if sp and (sp.Position - ro.Position).Magnitude < 18 then
                            nearPrompt = true; break
                        end
                    end
                end
                if nearPrompt then break end
            end
        end
        if nearPrompt then modeIndex = 2 else modeIndex = 1 end
        if lblMode then lblMode.Text = modeNames[modeIndex] end
    end

    local spd = currentSpeed()
    if toggles.SafeMode then spd = math.min(spd, 50) end

    local md  = h.MoveDirection
    local vel = Vector3.new(md.X * spd, ro.AssemblyLinearVelocity.Y, md.Z * spd)
    pcall(function() ro.AssemblyLinearVelocity = vel end)
    pcall(function() ro.Velocity = vel end)
end)

-- ── Ragdoll Timer + Ragdoll Steal ────────────────────────────
local function fireRagSteal()
    if not toggles.RagdollSteal then return end
    task.wait(0.05)
    local plots = WS:FindFirstChild("Plots"); if not plots then return end
    local ro = Root(); if not ro then return end
    local nearest, bestD = nil, math.huge
    for _, plot in ipairs(plots:GetChildren()) do
        local pods = plot:FindFirstChild("AnimalPodiums")
        if pods then
            for _, pod in ipairs(pods:GetChildren()) do
                local base = pod:FindFirstChild("Base"); local sp = base and base:FindFirstChild("Spawn")
                if sp then
                    local d = (sp.Position - ro.Position).Magnitude
                    local radius = tonumber(tbStealRadius and tbStealRadius.Text) or 54
                    if d <= radius and d < bestD then
                        local att = sp:FindFirstChild("PromptAttachment")
                        if att then
                            for _, prom in ipairs(att:GetChildren()) do
                                if prom:IsA("ProximityPrompt") and prom.ActionText and prom.ActionText:find("Steal") then
                                    nearest = prom; bestD = d
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    if nearest then
        local gc = getconn()
        if gc then
            local ok, tc = pcall(gc, nearest.Triggered)
            if ok and tc then for _, cn in ipairs(tc) do if cn.Function then task.spawn(cn.Function) end end end
        end
    end
end

local function triggerRagdollCountdown()
    if _ragRunning then return end
    _ragRunning = true
    if toggles.RagdollSteal then _ragStealArmed = true end
    task.spawn(function()
        for i = 3, 1, -1 do
            if not _ragWasIn and not _ragFired then
                clearRagDisplay(); _ragRunning = false; _ragStealArmed = false; return
            end
            if _ragTimerLabel then
                _ragTimerLabel.Text = tostring(i)
                _ragTimerLabel.TextColor3 = i == 1 and Color3.fromRGB(255, 105, 105) or Color3.fromRGB(255, 255, 255)
            end
            task.wait(0.9)
        end
        if _ragWasIn or _ragFired then
            if _ragTimerLabel then _ragTimerLabel.Text = "READY"; _ragTimerLabel.TextColor3 = Color3.fromRGB(100, 230, 100) end
            if _ragStealArmed then _ragStealArmed = false; task.spawn(fireRagSteal) end
        end
        _ragRunning = false; _ragFired = false
    end)
end

-- Ragdoll state watcher (shared between timer + steal + bat counter)
RunService.Heartbeat:Connect(function()
    local c = Char(); if not c then return end
    local h2 = c:FindFirstChildOfClass("Humanoid"); if not h2 or h2.Health <= 0 then return end
    local st = h2:GetState()
    local inRag = st == Enum.HumanoidStateType.Physics or st == Enum.HumanoidStateType.Ragdoll
        or st == Enum.HumanoidStateType.FallingDown

    -- Ragdoll Timer
    if toggles.RagdollTimer or toggles.RagdollSteal then
        if inRag then
            if not _ragWasIn then _ragWasIn = true; triggerRagdollCountdown() end
        else
            if _ragWasIn then
                _ragWasIn = false
                task.delay(1.5, function() if not _ragWasIn then clearRagDisplay() end end)
            end
        end
    end

    -- Bat Counter
    if toggles.BatCounter and inRag then
        if not _cbFired and not _cbDebounce then
            _cbFired = true; _cbDebounce = true
            task.spawn(function()
                local bat = findBat()
                if bat then
                    if bat.Parent ~= c then equip(bat); task.wait(0.05) end
                    pcall(function() bat:Activate() end)
                    task.wait(0.15); pcall(function() bat:Activate() end)
                end
                task.wait(0.5); _cbDebounce = false
            end)
        end
    end
    if not inRag then _cbFired = false end
end)

-- ── Medusa Counter ────────────────────────────────────────────
local function findMedusa()
    local c = Char(); local bp = LOCAL_PLAYER:FindFirstChild("Backpack")
    if c then for _, t in ipairs(c:GetChildren()) do if t:IsA("Tool") and t.Name:lower():find("medusa") then return t end end end
    if bp then for _, t in ipairs(bp:GetChildren()) do if t:IsA("Tool") and t.Name:lower():find("medusa") then return t end end end
end
local function useMedusa()
    if _medDebounce or tick() - _medLastUsed < MED_CD then return end
    local c = Char(); if not c then return end
    _medDebounce = true
    local med = findMedusa()
    if med then
        if med.Parent ~= c then equip(med) end
        pcall(function() med:Activate() end)
        _medLastUsed = tick()
        if toggles.AutoResetMedusa then task.delay(1.2, function() instaReset() end) end
    end
    _medDebounce = false
end
local function onPartAnchor(part)
    return part:GetPropertyChangedSignal("Anchored"):Connect(function()
        if toggles.MedusaCounter and part.Anchored and part.Transparency == 1 then useMedusa() end
    end)
end
function setupMedusa(char)
    teardownMedusa(); if not char then return end
    for _, p in ipairs(char:GetDescendants()) do
        if p:IsA("BasePart") then table.insert(_medusaConns, onPartAnchor(p)) end
    end
    table.insert(_medusaConns, char.DescendantAdded:Connect(function(p)
        if p:IsA("BasePart") then table.insert(_medusaConns, onPartAnchor(p)) end
    end))
end
function teardownMedusa()
    for _, c in ipairs(_medusaConns) do pcall(function() c:Disconnect() end) end; _medusaConns = {}
end

local function isMyPlot(plot)
    local sign = plot:FindFirstChild("PlotSign")
    if sign then
        local yb = sign:FindFirstChild("YourBase")
        if yb and yb:IsA("BillboardGui") and yb.Enabled then return true end
    end
    if plot.Name == LOCAL_PLAYER.Name or plot.Name == LOCAL_PLAYER.DisplayName then return true end
    local plots = WS:FindFirstChild("Plots")
    if plots then local pb = plots:FindFirstChild(LOCAL_PLAYER.Name); if pb and pb == plot then return true end end
    return false
end

local function findNearestPrompt()
    local c = LOCAL_PLAYER.Character; if not c then return end
    local root = c:FindFirstChild("HumanoidRootPart"); if not root then return end
    local plots = WS:FindFirstChild("Plots"); if not plots then return end
    local radius = tonumber(tbStealRadius and tbStealRadius.Text) or 54
    local nearest, bestD = nil, math.huge
    for _, plot in ipairs(plots:GetChildren()) do
        if not isMyPlot(plot) then
            local pods = plot:FindFirstChild("AnimalPodiums")
            if pods then
                for _, pod in ipairs(pods:GetChildren()) do
                    local base = pod:FindFirstChild("Base"); local sp = base and base:FindFirstChild("Spawn")
                    if sp then
                        local d = (sp.Position - root.Position).Magnitude
                        if d <= radius and d < bestD then
                            local att = sp:FindFirstChild("PromptAttachment")
                            if att then
                                for _, prom in ipairs(att:GetChildren()) do
                                    if prom:IsA("ProximityPrompt") and prom.ActionText and prom.ActionText:find("Steal") then
                                        nearest = prom; bestD = d
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    return nearest
end

local function executeSteal(prompt)
    if _isStealBusy then return end
    if not prompt or not prompt.Parent then return end
    if _ragStealArmed then return end  -- paused during ragdoll countdown
    if not _stealData[prompt] then
        _stealData[prompt] = {hold = {}, trigger = {}, ready = true}
        local gc = getconn()
        if gc then
            local ok1, hc = pcall(gc, prompt.PromptButtonHoldBegan)
            if ok1 and hc then for _, cn in ipairs(hc) do if cn.Function then table.insert(_stealData[prompt].hold, cn.Function) end end end
            local ok2, tc = pcall(gc, prompt.Triggered)
            if ok2 and tc then for _, cn in ipairs(tc) do if cn.Function then table.insert(_stealData[prompt].trigger, cn.Function) end end end
        end
    end
    local data = _stealData[prompt]; if not data.ready then return end
    data.ready = false; _isStealBusy = true
    task.spawn(function()
        for _, f in ipairs(data.hold) do pcall(f) end
        local t0 = tick()
        local dur = math.max(tonumber(nil) or 1.2, 0.5)
        while tick() - t0 < dur do updateProgressUI((tick()-t0)/dur); task.wait() end
        updateProgressUI(1)
        for _, f in ipairs(data.trigger) do pcall(f) end
        task.wait(0.05); updateProgressUI(0)
        data.ready = true; _isStealBusy = false
    end)
end

local _dropActive = false

    if _dropActive then return end
    _dropActive = true
    task.spawn(function()
        local ro = Root(); if not ro then _dropActive = false; return end
        local t0 = tick()
        local dc
        dc = RunService.Heartbeat:Connect(function()
            local r = Root(); if not r then dc:Disconnect(); _dropActive = false; return end

                dc:Disconnect()

                rp.FilterDescendantsInstances = {Char()}; rp.FilterType = Enum.RaycastFilterType.Exclude

                    r.AssemblyLinearVelocity = Vector3.zero; pcall(function() r.Velocity = Vector3.zero end)
                end
                _dropActive = false; return
            end
            r.AssemblyLinearVelocity = Vector3.new(r.AssemblyLinearVelocity.X, 150, r.AssemblyLinearVelocity.Z)
        end)
    end)
end

-- ── Bat Aimbot ───────────────────────────────────────────────
local _aimbotTarget = nil
local AIMBOT_HIT_DIST = 8
local _lastSwing = 0; local SWING_CD = 0.12

local function getTarget(myHRP)
    if _aimbotTarget and _aimbotTarget.Parent and _aimbotTarget.Parent:FindFirstChildOfClass("Humanoid") then
        return _aimbotTarget
    end
    _aimbotTarget = nil
    local best, bestD = nil, math.huge
    for _, pl in ipairs(Players:GetPlayers()) do
        if pl ~= LOCAL_PLAYER and pl.Character then
            local tr = pl.Character:FindFirstChild("HumanoidRootPart")
            local h2 = pl.Character:FindFirstChildOfClass("Humanoid")
            if tr and h2 and h2.Health > 0 then
                local d = (tr.Position - myHRP.Position).Magnitude
                if d < bestD then bestD = d; best = tr end
            end
        end
    end
    _aimbotTarget = best; return best
end

function startBatAimbot()
    if _aimbotConn then _aimbotConn:Disconnect(); _aimbotConn = nil end
    _aimbotConn = RunService.Heartbeat:Connect(function()
        if not toggles.BatAimbot then stopBatAimbot(); return end
        local c = Char(); if not c then return end
        local myHRP = c:FindFirstChild("HumanoidRootPart"); if not myHRP then return end
        local h2    = c:FindFirstChildOfClass("Humanoid"); if not h2 or h2.Health <= 0 then return end
        h2.AutoRotate = false

        local bat = findBat()
        if bat and bat.Parent ~= c then pcall(function() h2:EquipTool(bat) end) end

        local target = getTarget(myHRP)
        if not target or not target.Parent then
            h2.AutoRotate = true; myHRP.AssemblyLinearVelocity = Vector3.zero; return
        end

        local tVel = target.AssemblyLinearVelocity
        local pred = target.Position + tVel * math.clamp(tVel.Magnitude / 150, 0.05, 0.2)

        local batSpeed
        if toggles.BatSpeed then
            batSpeed = laggerOn
                and (tonumber(tbBatLaggerSpeed and tbBatLaggerSpeed.Text) or 65)
                or  (tonumber(tbBatNormalSpeed and tbBatNormalSpeed.Text) or 58)
        else
            batSpeed = currentSpeed()
        end

        local behindOffset  = -target.CFrame.LookVector * 2.5
        local heightOffset  = Vector3.new(0, 2.8, 0)
        local standPos      = pred + behindOffset + heightOffset
        local moveDir       = standPos - myHRP.Position
        local flatDir       = Vector3.new(moveDir.X, 0, moveDir.Z)

        if aimbotType == "DEFAULT" then
            -- Smooth arc approach
            if moveDir.Magnitude > 1 then
                myHRP.AssemblyLinearVelocity = moveDir.Unit * batSpeed
            else
                myHRP.AssemblyLinearVelocity = tVel
            end
        else
            -- BYPASS: direct velocity + auto-rotate
            if flatDir.Magnitude > 0.1 then flatDir = flatDir.Unit else flatDir = Vector3.zero end
            local yVel = (pred.Y + 3.5 - myHRP.Position.Y) * 20 + tVel.Y * 0.8
            yVel = math.clamp(yVel, -70, 110)
            local desired = Vector3.new(flatDir.X * batSpeed, yVel, flatDir.Z * batSpeed)
            myHRP.AssemblyLinearVelocity = myHRP.AssemblyLinearVelocity:Lerp(desired, 0.8)
        end

        -- Look at target
        if flatDir.Magnitude > 0.1 then
            local lookF = Vector3.new(moveDir.X, 0, moveDir.Z)
            if lookF.Magnitude > 0.1 then
                local newCF = CFrame.lookAt(myHRP.Position, myHRP.Position + lookF)
                myHRP.CFrame = CFrame.new(myHRP.Position) * (newCF - newCF.Position)
            end
        end

        -- Auto swing
        local dist = (target.Position - myHRP.Position).Magnitude
        if dist <= AIMBOT_HIT_DIST then
            local now = tick()
            if now - _lastSwing >= SWING_CD then _lastSwing = now; task.spawn(swingBat) end
        end
    end)
end
function stopBatAimbot()
    if _aimbotConn then _aimbotConn:Disconnect(); _aimbotConn = nil end
    _aimbotTarget = nil
    local c = Char(); if not c then return end
    local h2 = c:FindFirstChildOfClass("Humanoid"); if h2 then h2.AutoRotate = true end
    local ro = c:FindFirstChild("HumanoidRootPart"); if ro then ro.AssemblyLinearVelocity = Vector3.zero end
end

                -- Immediate counter: equip bat + swing back
                local bat = findBat()
                if bat then
                    if bat.Parent ~= c then equip(bat); task.wait(0.05) end
                    pcall(function() bat:Activate() end); task.wait(0.1); pcall(function() bat:Activate() end)
                end
                -- Recover
                pcall(function() h2:ChangeState(Enum.HumanoidStateType.Running) end)
                local ro = c:FindFirstChild("HumanoidRootPart")
                if ro then ro.AssemblyLinearVelocity = Vector3.zero end
                task.wait(0.4); _lastCB = false
            end)
        end
    end)
end
function stopAntiBat()
    if _antiBatConn then _antiBatConn:Disconnect(); _antiBatConn = nil end
end

-- One-shot Anti Bat (V key)
local function antiBatOneShot()
    local bat = findBat(); if not bat then return end

    if bat.Parent ~= c then equip(bat); task.wait(0.05) end
    pcall(function() bat:Activate() end)
end

-- ── Anti Lag ─────────────────────────────────────────────────
local function antiLagDerender(obj)
    pcall(function()
        if obj:IsA("Accessory") or obj:IsA("Hat") then obj:Destroy()
        elseif obj:IsA("BasePart") then obj.Material = Enum.Material.Plastic; obj.Reflectance = 0; obj.CastShadow = false
        elseif obj:IsA("Decal") or obj:IsA("Texture") then obj.Transparency = 1
        elseif obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam")
            or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then obj.Enabled = false
        end
    end)
end
function enableAntiLag()
    _defBright  = _defBright  or Lighting.Brightness
    _defClock   = _defClock   or Lighting.ClockTime
    _defAmbient = _defAmbient or Lighting.OutdoorAmbient
    Lighting.GlobalShadows = false; Lighting.FogEnd = 1e10; Lighting.Brightness = 1
    Lighting.EnvironmentDiffuseScale = 0; Lighting.EnvironmentSpecularScale = 0
    for _, e in pairs(Lighting:GetChildren()) do
        pcall(function()
            if e:IsA("BlurEffect") or e:IsA("SunRaysEffect") or e:IsA("ColorCorrectionEffect")
                or e:IsA("BloomEffect") or e:IsA("DepthOfFieldEffect") then e.Enabled = false end
        end)
    end
    for _, obj in ipairs(WS:GetDescendants()) do antiLagDerender(obj) end
    if _antiLagDescConn then _antiLagDescConn:Disconnect() end
    _antiLagDescConn = WS.DescendantAdded:Connect(function(obj)
        if toggles.AntiLag then antiLagDerender(obj) end
    end)
end
function disableAntiLag()
    if _antiLagDescConn then _antiLagDescConn:Disconnect(); _antiLagDescConn = nil end
    pcall(function()
        if _defBright  then Lighting.Brightness    = _defBright  end
        if _defClock   then Lighting.ClockTime      = _defClock   end
        if _defAmbient then Lighting.OutdoorAmbient = _defAmbient end
        Lighting.GlobalShadows = true
    end)
end

-- ── Save Config ──────────────────────────────────────────────
function saveConfig()
    local ok, _ = pcall(function()
        local data = {
            toggles     = toggles,
            modeIndex   = modeIndex,
            laggerOn    = laggerOn,
            jumpMethod  = jumpMethod,
            aimbotType  = aimbotType,
            selectedMusic = selectedMusic,
            speeds = {
                NormalSpeed      = tonumber(tbNormalSpeed     and tbNormalSpeed.Text)     or 59,
                CarrySpeed       = tonumber(tbCarrySpeed      and tbCarrySpeed.Text)      or 28,
                LaggerSpeed      = tonumber(tbLaggerSpeed     and tbLaggerSpeed.Text)     or 15,
                LaggerCarrySpeed = tonumber(tbLaggerCarrySpeed and tbLaggerCarrySpeed.Text) or 24.5,
                StealRadius      = tonumber(tbStealRadius     and tbStealRadius.Text)     or 54,
                TPDownHeight     = tonumber(tbTPDownHeight    and tbTPDownHeight.Text)    or 20,
                BatNormalSpeed   = tonumber(tbBatNormalSpeed  and tbBatNormalSpeed.Text)  or 58,
                BatLaggerSpeed   = tonumber(tbBatLaggerSpeed  and tbBatLaggerSpeed.Text)  or 65,
                UIScale          = tonumber(tbUIScale         and tbUIScale.Text)         or 90,
            },
        }
        writefile("envy_config.json", HttpService:JSONEncode(data))
    end)
    if not ok then warn("[EnvyHub] Save unavailable in this executor") end
end

local function loadConfig()
    pcall(function()
        local ok, raw = pcall(readfile, "envy_config.json")
        if not ok or not raw or raw == "" then return end
        local data = HttpService:JSONDecode(raw)
        if data.speeds then
            if tbNormalSpeed      and data.speeds.NormalSpeed      then tbNormalSpeed.Text      = tostring(data.speeds.NormalSpeed)      end
            if tbCarrySpeed       and data.speeds.CarrySpeed       then tbCarrySpeed.Text       = tostring(data.speeds.CarrySpeed)       end
            if tbLaggerSpeed      and data.speeds.LaggerSpeed      then tbLaggerSpeed.Text      = tostring(data.speeds.LaggerSpeed)      end
            if tbLaggerCarrySpeed and data.speeds.LaggerCarrySpeed then tbLaggerCarrySpeed.Text = tostring(data.speeds.LaggerCarrySpeed) end
            if tbStealRadius      and data.speeds.StealRadius      then tbStealRadius.Text      = tostring(data.speeds.StealRadius)      end
            if tbTPDownHeight     and data.speeds.TPDownHeight     then tbTPDownHeight.Text     = tostring(data.speeds.TPDownHeight)     end
            if tbBatNormalSpeed   and data.speeds.BatNormalSpeed   then tbBatNormalSpeed.Text   = tostring(data.speeds.BatNormalSpeed)   end
            if tbBatLaggerSpeed   and data.speeds.BatLaggerSpeed   then tbBatLaggerSpeed.Text   = tostring(data.speeds.BatLaggerSpeed)   end
            if tbUIScale          and data.speeds.UIScale          then tbUIScale.Text          = tostring(data.speeds.UIScale)          end
        end
        if data.modeIndex   then modeIndex   = data.modeIndex;   if lblMode        then lblMode.Text        = modeNames[modeIndex] end end
        if data.jumpMethod  then jumpMethod  = data.jumpMethod;  if lblJumpMethod  then lblJumpMethod.Text  = jumpMethod           end end
        if data.aimbotType  then aimbotType  = data.aimbotType;  if lblAimbotType  then lblAimbotType.Text  = aimbotType           end end
        if data.selectedMusic then selectedMusic = data.selectedMusic end
        if data.toggles then
            for k, v in pairs(data.toggles) do
                if toggles[k] ~= nil then setToggle(k, v) end
            end
        end
    end)
end

-- // [8] KEYBIND SYSTEM // --

local function setupKeybindButton(btn, action)
    local binding = false
    btn.MouseButton1Click:Connect(function()
        if binding then return end
        binding = true
        local prev = btn.Text
        btn.Text = "..."
        local conn; conn = UserInputService.InputBegan:Connect(function(inp, gp)
            if gp then return end
            if inp.UserInputType == Enum.UserInputType.Keyboard then
                keybinds[action] = inp.KeyCode
                btn.Text = inp.KeyCode.Name
                binding = false; conn:Disconnect()
            end
        end)
        task.delay(8, function()  -- timeout
            if binding then binding = false; btn.Text = prev; pcall(function() conn:Disconnect() end) end
        end)
    end)
end

setupKeybindButton(kbMode,      "mode")
setupKeybindButton(kbLagger,    "lagger")
setupKeybindButton(kbAutoLeft,  "autoLeft")
setupKeybindButton(kbAutoRight, "autoRight")
setupKeybindButton(kbDrop,      "drop")
setupKeybindButton(kbTPFloor,   "tpFloor")
setupKeybindButton(kbBatAimbot, "batAimbot")
setupKeybindButton(kbAntiBat,   "antiBat")
setupKeybindButton(kbReset,     "reset")
setupKeybindButton(kbHideShow,  "hideShow")

-- Keybind action handler
UserInputService.InputBegan:Connect(function(inp, gp)
    if gp then return end
    if UserInputService:GetFocusedTextBox() then return end
    local kc = inp.KeyCode

    if kc == keybinds.mode then
        modeIndex = (modeIndex % #modeNames) + 1
        if lblMode then lblMode.Text = modeNames[modeIndex] end

    elseif kc == keybinds.lagger then
        laggerOn = not laggerOn
        if laggerOn then
            if modeIndex == 1 then modeIndex = 3 elseif modeIndex == 2 then modeIndex = 4 end
        else
            if modeIndex == 3 then modeIndex = 1 elseif modeIndex == 4 then modeIndex = 2 end
        end
        if lblMode then lblMode.Text = modeNames[modeIndex] end

    elseif kc == keybinds.autoLeft then
        local s = not toggles.AutoLeft; setToggle("AutoLeft", s)
        if s then startAutoLeft() else stopAutoLeft() end

    elseif kc == keybinds.autoRight then
        local s = not toggles.AutoRight; setToggle("AutoRight", s)
        if s then startAutoRight() else stopAutoRight() end

    elseif kc == keybinds.drop then
        doDrop()

    elseif kc == keybinds.tpFloor then
        doTPFloor()

    elseif kc == keybinds.batAimbot then
        local s = not toggles.BatAimbot; setToggle("BatAimbot", s)
        if s then startBatAimbot() else stopBatAimbot() end

    elseif kc == keybinds.antiBat then
        if toggles.AntiBat then
            antiBatOneShot()
        else
            local s = not toggles.AntiBat; setToggle("AntiBat", s)
            if s then startAntiBat() else stopAntiBat() end
        end

    elseif kc == keybinds.reset then
        instaReset()

    elseif kc == keybinds.hideShow then
        isGUIVisible = not isGUIVisible
        Main.Visible = isGUIVisible and not isMinimized
        EnvyMini.Visible = not isGUIVisible
    end
end)
